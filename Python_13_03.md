```
n = int(input())    #Ввести через панель значение.
r = bin(n)[2:]      #Перевод введёного значения в число с основанием 2 убирая первые два символа.
if n % 2 == 0:      #Если значение n делится на 2 без остатка то,
    r += '01'       #оно чётное и приписываем справа 01
else:               #или
    r += '10'       #оно нечётное и припысываем 10.
print(r)            #Выводим полученное значение.
```

```
for n in range(1, 100):         #Производим перебор чисел от 1 до 99.
    r = bin(n)[2:]              #Перевод значени n в число с основанием 2 убирая первые два символа.
    if n % 2 == 0:              #Если значение n делится на 2 без остатка то,
        r += '01'               #оно чётное и приписываем справа 01
    else:                       #или
        r += '10'               #оно не нечётное и припысываем 10.
    if int(r, 2) > 130:         #Если полученное значение больше 130 то,
        print(int(r, 2))        #Выводим его и переводим десятичную.
```
